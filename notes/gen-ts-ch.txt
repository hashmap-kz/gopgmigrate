package main

import (
	"bytes"
	"fmt"
	"log"
	"os"
	"path/filepath"
	"strings"
	"text/template"
	"time"
)

type MigrationFile struct {
	Dir      string
	Filename string
	Content  string
}

type MigrationFilePair struct {
	Do   MigrationFile
	Undo MigrationFile
}

func ExecTemplate(name, t string, data map[string]string) string {
	var result bytes.Buffer
	tmpl, err := template.New(name).Parse(t)
	if err != nil {
		log.Fatal(err)
	}
	err = tmpl.Execute(&result, data)
	if err != nil {
		log.Fatal(err)
	}
	return result.String()
}

var funcTemplate = strings.TrimSpace(`
drop function if exists {{.schema}}.fn_get_{{.table}};
create function {{.schema}}.fn_get_{{.table}}()
    returns table
            (
                id    int,
                xname text
            )
as
$fn$
begin
    return query
        select c.id   as id,
               c.xname as xname
        from {{.table}} c;
end
$fn$ language plpgsql;
`)

var alterTmplDo = strings.TrimSpace(`
alter table {{.schema}}.{{.table}} add column xdata text;
update {{.schema}}.{{.table}} set xdata = xname || '-data';
create index ix_{{.table}}_xdata on {{.schema}}.{{.table}} (xdata);
`)

var alterTmplUndo = strings.TrimSpace(`
alter table {{.schema}}.{{.table}} drop column xdata;
`)

var viewTmpl = strings.TrimSpace(`
create view {{.schema}}.vw_{{.table}} as
select * from {{.schema}}.{{.table}};
`)

func w(filePath, content string) {
	if err := os.MkdirAll(filepath.Dir(filePath), os.ModePerm); err != nil {
		log.Fatal(err)
	}
	if err := os.WriteFile(filePath, []byte(content), 0644); err != nil {
		log.Fatal(err)
	}
	fmt.Println("Generated:", filePath)
}

func one(tmpl MigrationFile, fileNumber, schemaName, tableName, targetDir string) {
	filename := fmt.Sprintf(tmpl.Filename, fileNumber, tableName)
	filePath := filepath.Join(targetDir, tmpl.Dir, filename)
	if tmpl.Dir != "undo" {
		if tmpl.Dir == "schema" || tmpl.Dir == "data" {
			filePath = filepath.Join(targetDir, tmpl.Dir, schemaName, tableName, filename)
		} else {
			filePath = filepath.Join(targetDir, tmpl.Dir, schemaName, filename)
		}
	}
	content := ExecTemplate(tableName, tmpl.Content, map[string]string{
		"schema": schemaName,
		"table":  tableName,
		"idx":    fmt.Sprintf("%s", fileNumber),
	})
	w(filePath, content)
}

func genTimestamps() []string {
	const cnt = 1000

	startTime := time.Now().UTC()
	timestamps := make([]string, cnt)

	for i := 0; i < cnt; i++ {
		timestamps[i] = startTime.Add(time.Duration(i) * time.Second).Format("20060102150405")
	}

	return timestamps
}

func main() {
	targetDir := "./migrations"
	os.RemoveAll(targetDir)

	// Ensure the target directory exists
	if err := os.MkdirAll(targetDir, os.ModePerm); err != nil {
		fmt.Println("Error creating directory:", err)
		return
	}

	numbers := genTimestamps()
	getNextSeq := func() string {
		if len(numbers) > 0 {
			first := numbers[0]
			numbers = numbers[1:]
			return first
		}
		log.Fatalf("out of bounds")
		return ""
	}

	schemas := []string{
		"authn",
		"public",
	}

	tables := []string{
		"authn.users",
		"authn.roles",
		"authn.permissions",
		"authn.groups",
		"public.companies",
	}

	filesTemplate := []MigrationFilePair{
		{
			Do:   MigrationFile{Dir: "schema", Filename: "%s-%s_table.do.sql", Content: `create table if not exists {{.schema}}.{{.table}} (id serial primary key, xname text not null);`},
			Undo: MigrationFile{Dir: "undo", Filename: "%s-%s_table.undo.sql", Content: `drop table if exists {{.schema}}.{{.table}};`},
		},
		{
			MigrationFile{Dir: "repeatable/functions", Filename: "%s-fn_get_%s.r.sql", Content: funcTemplate},
			MigrationFile{Dir: "undo", Filename: "%s-fn_get_%s.undo.sql", Content: `drop function if exists {{.schema}}.fn_get_{{.table}};`},
		},
		{
			MigrationFile{Dir: "repeatable/views", Filename: "%s-vw_%s.r.sql", Content: viewTmpl},
			MigrationFile{Dir: "undo", Filename: "%s-vw_%s.undo.sql", Content: `drop view if exists {{.schema}}.vw_{{.table}};`},
		},
		{
			MigrationFile{Dir: "data", Filename: "%s-seed_%s.r.sql", Content: `insert into {{.schema}}.{{.table}} (xname) values ('test-{{.idx}}');`},
			MigrationFile{Dir: "undo", Filename: "%s-seed_%s.undo.sql", Content: `delete from {{.schema}}.{{.table}} where xname = 'test-{{.idx}}';`},
		},
		{
			MigrationFile{Dir: "schema", Filename: "%s-alter_%s_add_xdata.do.sql", Content: alterTmplDo},
			MigrationFile{Dir: "undo", Filename: "%s-alter_%s_add_xdata.undo.sql", Content: alterTmplUndo},
		},
	}

	for _, elem := range schemas {
		fileNumber := getNextSeq()
		filenameDo := fmt.Sprintf("%s-%s_schema.do.sql", fileNumber, elem)
		filenameUndo := fmt.Sprintf("%s-%s_schema.undo.sql", fileNumber, elem)

		w(filepath.Join(targetDir, "schema", filenameDo),
			ExecTemplate(elem, `create schema if not exists {{.schema}};`, map[string]string{
				"schema": elem,
			}))
		w(filepath.Join(targetDir, "undo", filenameUndo),
			ExecTemplate(elem, `create schema if not exists {{.schema}};`, map[string]string{
				"schema": elem,
			}))
	}

	for _, schTab := range tables {
		schemaName := strings.Split(schTab, ".")[0]
		tableName := strings.Split(schTab, ".")[1]
		for _, tmpl := range filesTemplate {
			fileNumber := getNextSeq()
			one(tmpl.Do, fileNumber, schemaName, tableName, targetDir)
			one(tmpl.Undo, fileNumber, schemaName, tableName, targetDir)
		}
	}

	fmt.Println("Test files generated successfully!")
}
