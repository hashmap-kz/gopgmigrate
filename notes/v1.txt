package main

import (
	"context"
	"crypto/sha256"
	"encoding/hex"
	"fmt"
	"log"
	"os"
	"path/filepath"
	"sort"
	"strconv"
	"strings"

	"github.com/jackc/pgx/v5"
)

// GetDatabaseConnection initializes a PostgreSQL connection
func GetDatabaseConnection(dbURL string) (*pgx.Conn, error) {
	ctx := context.Background()
	conn, err := pgx.Connect(ctx, dbURL)
	return conn, err
}

// Ensure migration tracking tables exist
func EnsureSchemaMigrationTables(conn *pgx.Conn) error {
	query := `
		create table if not exists schema_migrations (
			id 			serial primary key,
			version 	varchar(8) unique not null,
			applied_at 	timestamp default now()
		);
		create table if not exists schema_migrations_repeatable (
			id 			serial primary key,
			name 		text unique not null,
			hash 		text not null,
			applied_at 	timestamp default now()
		);
	`
	_, err := conn.Exec(context.Background(), query)
	return err
}

// Get applied migrations
func GetAppliedMigrations(conn *pgx.Conn) (map[int]bool, error) {
	rows, err := conn.Query(context.Background(), "SELECT version FROM schema_migrations")
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	migrations := make(map[int]bool)
	for rows.Next() {
		var version int
		if err := rows.Scan(&version); err != nil {
			return nil, err
		}
		migrations[version] = true
	}
	return migrations, nil
}

// Get applied repeatable migrations and their hashes
func GetAppliedRepeatableMigrations(conn *pgx.Conn) (map[string]string, error) {
	rows, err := conn.Query(context.Background(), "SELECT name, hash FROM schema_migrations_repeatable")
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	migrations := make(map[string]string)
	for rows.Next() {
		var name, hash string
		if err := rows.Scan(&name, &hash); err != nil {
			return nil, err
		}
		migrations[name] = hash
	}
	return migrations, nil
}

// Compute SHA256 hash of a file
func ComputeHash(content []byte) string {
	hash := sha256.Sum256(content)
	return hex.EncodeToString(hash[:])
}

// Apply migration (up/down)
func ApplyMigration(conn *pgx.Conn, version int, direction, sql string) error {
	ctx := context.Background()
	tx, err := conn.Begin(ctx)
	if err != nil {
		return err
	}
	defer tx.Rollback(context.Background())

	_, err = tx.Exec(ctx, sql)
	if err != nil {
		return err
	}

	if direction == "up" {
		_, err = tx.Exec(ctx, "INSERT INTO schema_migrations (version) VALUES ($1)", version)
	} else {
		_, err = tx.Exec(ctx, "DELETE FROM schema_migrations WHERE version = $1", version)
	}

	if err != nil {
		return err
	}

	return tx.Commit(ctx)
}

// Apply repeatable migration (if changed)
func ApplyRepeatableMigration(conn *pgx.Conn, name, sql string, newHash string) error {
	ctx := context.Background()

	tx, err := conn.Begin(ctx)
	if err != nil {
		return err
	}

	_, err = tx.Exec(ctx, sql)
	if err != nil {
		tx.Rollback(ctx)
		return err
	}

	_, err = tx.Exec(ctx, `
		INSERT INTO schema_migrations_repeatable (name, hash)
		VALUES ($1, $2)
		ON CONFLICT (name) DO UPDATE SET hash = $2, applied_at = NOW()`, name, newHash,
	)

	if err != nil {
		tx.Rollback(ctx)
		return err
	}

	return tx.Commit(ctx)
}

// Run versioned migrations
func RunMigrations(conn *pgx.Conn, folder, direction string) error {
	applied, err := GetAppliedMigrations(conn)
	if err != nil {
		return err
	}

	files, err := filepath.Glob(filepath.Join(folder, "*.up.sql"))
	if err != nil {
		return err
	}
	sort.Strings(files)

	for _, file := range files {
		if !strings.HasSuffix(file, fmt.Sprintf(".%s.sql", direction)) {
			continue
		}

		versionStr := strings.Split(filepath.Base(file), "_")[0]
		version, err := strconv.Atoi(versionStr)
		if err != nil {
			return fmt.Errorf("invalid migration filename format: %s", file)
		}

		if direction == "up" && applied[version] {
			continue
		}
		if direction == "down" && !applied[version] {
			continue
		}

		sql, err := os.ReadFile(file)
		if err != nil {
			return err
		}

		fmt.Printf("Applying migration %s...\n", file)
		err = ApplyMigration(conn, version, direction, string(sql))
		if err != nil {
			return err
		}
	}

	return nil
}

// Run repeatable migrations
func RunRepeatableMigrations(conn *pgx.Conn, folder string) error {
	applied, err := GetAppliedRepeatableMigrations(conn)
	if err != nil {
		return err
	}

	files, err := filepath.Glob(filepath.Join(folder, "*.repeatable.sql"))
	if err != nil {
		return err
	}

	for _, file := range files {
		sql, err := os.ReadFile(file)
		if err != nil {
			return err
		}

		newHash := ComputeHash(sql)
		name := filepath.Base(file)

		// Only apply if new or changed
		if applied[name] != newHash {
			fmt.Printf("Applying repeatable migration %s...\n", file)
			err = ApplyRepeatableMigration(conn, name, string(sql), newHash)
			if err != nil {
				return err
			}
		}
	}

	return nil
}

func main() {
	conn, _ := GetDatabaseConnection("postgres://postgres:postgres@localhost:5432/bookstore")
	defer conn.Close(context.Background())

	err := EnsureSchemaMigrationTables(conn)
	if err != nil {
		log.Fatal(err)
	}

	if len(os.Args) < 2 {
		fmt.Println("Usage: go run main.go [up|down|repeatable]")
		os.Exit(1)
	}

	direction := os.Args[1]

	if direction == "repeatable" {
		err := RunRepeatableMigrations(conn, "migrations")
		if err != nil {
			log.Fatal(err)
		}
	} else {
		err := RunMigrations(conn, "migrations", direction)
		if err != nil {
			log.Fatal(err)
		}
	}

	fmt.Println("Migrations applied successfully!")
}
