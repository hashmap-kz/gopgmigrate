package arch

import "gopgmigrate/internal/migrate"

func FindCorrespondingUndoScript(undoScripts []migrate.MigrationFile, doScript migrate.MigrationFile) (*migrate.MigrationFile, bool, error) {
	versionDo, err := migrate.parseVersionDo(doScript.Base)
	if err != nil {
		return nil, false, err
	}
	for _, elem := range undoScripts {
		versionUndo, err := migrate.parseVersionUndo(elem.Base)
		if err != nil {
			return nil, false, err
		}
		if versionUndo == versionDo {
			return &elem, true, nil
		}
	}
	return nil, false, nil
}
