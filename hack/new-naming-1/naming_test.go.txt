func TestParseVersionDo(t *testing.T) {
	tests := []struct {
		filename string
		expected int64
		hasError bool
	}{
		{"0000001-users.do.sql", 1, false},
		{"0012345-roles.notx.do.sql", 12345, false},
		{"0000000-init.r.do.sql", 0, false},
		{"0000123-test.rnotx.do.sql", 123, false},
		{"9999999-complex.name [v2] .do.sql", 9999999, false},

		{"123456-users.do.sql", -1, true},      // Invalid: needs exactly 7 digits
		{"000001-users.do.sql", -1, true},      // Invalid: only 6 digits
		{"0000001_users.do.sql", -1, true},     // Invalid: uses `_` instead of `-` after version
		{"0000001-users.up.sql", -1, true},     // Invalid: wrong suffix
		{"users.do.sql", -1, true},             // Invalid: missing version number
		{"0000001-users.sql", -1, true},        // Invalid: missing known do suffix
		{"0000001-users.undo.sql", -1, true},   // Invalid: undo is not a do migration
		{"0000001-users.do.sql.bak", -1, true}, // Invalid: additional suffix
		{"0000001-dir/users.do.sql", -1, true}, // Invalid: slash in basename
		{"0000001-.do.sql", -1, true},          // Invalid: empty name
	}

	for _, test := range tests {
		t.Run(test.filename, func(t *testing.T) {
			result, err := ParseVersionDo(test.filename)

			if test.hasError {
				assert.Error(t, err)
				assert.Equal(t, int64(-1), result)
			} else {
				require.NoError(t, err)
				assert.Equal(t, test.expected, result)
			}
		})
	}
}

func TestParseMigrationFilename(t *testing.T) {
	tests := []struct {
		input    string
		want     ParsedMigration
		hasError bool
	}{
		{
			input: "0000001-users.do.sql",
			want:  ParsedMigration{Version: 1, Name: "users", Kind: MigrationKindDo},
		},
		{
			input: "0000002-vacuum users.notx.do.sql",
			want:  ParsedMigration{Version: 2, Name: "vacuum users", Kind: MigrationKindNotxDo},
		},
		{
			input: "0000003-fn.create.user.r.do.sql",
			want:  ParsedMigration{Version: 3, Name: "fn.create.user", Kind: MigrationKindRDo},
		},
		{
			input: "0000004-name.with.undo.inside.rnotx.do.sql",
			want:  ParsedMigration{Version: 4, Name: "name.with.undo.inside", Kind: MigrationKindRNotx},
		},
		{
			input: "0000005-name.with.r.do.inside.undo.sql",
			want:  ParsedMigration{Version: 5, Name: "name.with.r.do.inside", Kind: MigrationKindUndo},
		},
		{input: "0000006-.undo.sql", hasError: true},
		{input: "0000006-dir/file.do.sql", hasError: true},
		{input: "0000006-file.ntx.do.sql", hasError: true},
		{input: "0000006-file.r.sql", hasError: true},
		{input: "abcdefg-file.do.sql", hasError: true},
	}

	for _, tt := range tests {
		t.Run(tt.input, func(t *testing.T) {
			got, err := ParseMigrationFilename(tt.input)
			if tt.hasError {
				assert.Error(t, err)
				return
			}
			require.NoError(t, err)
			assert.Equal(t, tt.want, got)
		})
	}
}

func TestVersionedMigrationRegexDo(t *testing.T) {
	tests := []struct {
		input   string
		matches bool
		version string
		name    string
		migType string
	}{
		{"0000003-users.do.sql", true, "0000003", "users", "do"},
		{"0000004-fn_list_users.r.do.sql", false, "", "", ""},
		{"0000004-name.with.dots.do.sql", true, "0000004", "name.with.dots", "do"},
		{"123-invalid.sql", false, "", "", ""},
		{"0000006-wrong.do.txt", false, "", "", ""},
	}

	for _, test := range tests {
		t.Run(test.input, func(t *testing.T) {
			matches := versionedMigrationRegexDo.FindStringSubmatch(test.input)
			assert.Equal(t, test.matches, matches != nil)

			if test.matches {
				assert.Equal(t, test.version, matches[1])
				assert.Equal(t, test.name, matches[2])
				assert.Equal(t, test.migType, matches[3])
			}
		})
	}
}

func TestHelpers(t *testing.T) {
	tests := []struct {
		base         string
		isTx         bool
		isRepeatable bool
		isVersioned  bool
		isUndo       bool
		isNonTx      bool
	}{
		{"0000001-users.do.sql", true, false, true, false, false},
		{"0000001-users.notx.do.sql", false, false, true, false, true},
		{"0000001-users.r.do.sql", true, true, true, false, false},
		{"0000001-users.rnotx.do.sql", false, true, true, false, true},
		{"0000001-users.undo.sql", true, false, false, true, false},
		{"broken.sql", false, false, false, false, false},
	}

	for _, tt := range tests {
		t.Run(tt.base, func(t *testing.T) {
			file := MigrationFile{Base: tt.base}
			assert.Equal(t, tt.isTx, IsTx(file))
			assert.Equal(t, tt.isRepeatable, IsRepeatable(file))
			assert.Equal(t, tt.isVersioned, IsVersioned(tt.base))
			assert.Equal(t, tt.isUndo, IsUndo(tt.base))
			assert.Equal(t, tt.isNonTx, IsNonTransaction(tt.base))
		})
	}
}

func TestIsOurRegex(t *testing.T) {
	assert.True(t, IsOurRegex(versionedMigrationRegexDo))
	assert.True(t, IsOurRegex(versionedMigrationRegexUndo))
	assert.True(t, IsOurRegex(repeatableMigrationRegexDo))
	assert.True(t, IsOurRegex(versionedMigrationRegexNotx))
	assert.True(t, IsOurRegex(MigrationRegex()))
	assert.False(t, IsOurRegex(regexp.MustCompile(`^abc$`)))
}

func TestParseVersionByRegex(t *testing.T) {
	version, err := ParseVersionByRegex("0000007-users.do.sql", versionedMigrationRegexDo)
	require.NoError(t, err)
	assert.Equal(t, int64(7), version)

	_, err = ParseVersionByRegex("0000007-users.do.sql", regexp.MustCompile(`^abc$`))
	assert.Error(t, err)

	_, err = ParseVersionByRegex("broken", versionedMigrationRegexDo)
	assert.Error(t, err)
}

func TestPostgresqlSchemaTablePathRegex(t *testing.T) {
	tests := []struct {
		input   string
		matches bool
	}{
		{"myschema.table", true},
		{"m$yschema1.m$table", true},
		{"public.users", true},
		{"my_schema.table$", true},
		{"sche$ma.table_123$456", true},

		{"123schema.table", false},
		{"myschema..table", false},
		{"myschema-table", false},
		{"_schema.$table", false},
		{"schema.123table", false},
		{"schema..table", false},
		{".table", false},
		{"schema.", false},
		{"schema.table$@", false},
		{"schema.reallylongtablename111111111111$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$", false},
	}

	for _, test := range tests {
		t.Run(test.input, func(t *testing.T) {
			matches := postgresqlSchemaTablePathRegex.MatchString(test.input)
			assert.Equal(t, test.matches, matches)
		})
	}
}
