// readString handles both ClickHouse and PostgreSQL string formats
func (t *Tokenizer) readString(stop rune) string {
	var sb strings.Builder
	sb.WriteRune(stop)
	for {
		r, _, ok := t.nextRune()
		if !ok {
			break
		}
		sb.WriteRune(r)

		if r == stop {
			// Handle escaped single quotes ('') for PostgreSQL
			if next, _, ok := t.peekRune(); ok && next == stop {
				t.nextRune()
				sb.WriteRune(next)
			} else {
				break
			}
		} else if r == '\\' { // Handle backslash escaping for ClickHouse
			if next, _, ok := t.peekRune(); ok && (next == stop || next == '\\') {
				t.nextRune()
				sb.WriteRune(next)
			}
		}
	}
	return sb.String()
}
